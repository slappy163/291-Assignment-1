CREATE TABLE Users(
     uid CHAR(10),
     name CHAR(20),
     PRIMARY KEY (uid)
)

CREATE TABLE create_playlist(
     uid CHAR(10) NOT NULL,
     plid CHAR(20),
     title CHAR(20),
     PRIMARY KEY (plid),
     FOREIGN KEY (uid) REFERENCES Users,
     FOREIGN KEY (plid) REFERENCES Playlists
)

CREATE TABLE pl_include(
     plid CHAR(20),
     sid CHAR(10),
     order CHAR(10), 
     PRIMARY KEY (plid, sid),
     FOREIGN KEY (sid) REFERENCES Songs,
     FOREIGN KEY (plid) REFERENCES Playlist
)

CREATE TABLE Songs(
     sid CHAR(10), 
     title CHAR(50),
     duration CHAR(10),
     PRIMARY KEY (sid)
)

CREATE TABLE Perform(
     sid CHAR(10),
     aid CHAR(11) NOT NULL,
     PRIMARY KEY (sid, aid),
     FOREIGN KEY (sid) REFERENCES Songs,
     FOREIGN KEY (aid) REFERENCES Artists
)

CREATE TABLE Artists(
     ain CHAR(11),
     name CHAR(50),
     PRIMARY KEY (ain)
)

CREATE TABLE follow_a(
     aid CHAR(11),
     uid CHAR(10),
     PRIMARY KEY (sid, uid),
     FOREIGN KEY (sid) REFERENCES Artists,
     FOREIGN KEY (uid) REFERENCES Users
)

CREATE TABLE followee(
     uid CHAR(10),
     PRIMARY KEY (uid),
     FOREIGN KEY (uid) REFERENCES Users
)

CREATE TABLE follower(
     uid CHAR(10),
     PRIMARY KEY (uid),
     FOREIGN KEY (uid) REFERENCES Users
)

CREATE TABLE Albums(
     albid CHAR(20), 
     title CHAR(20),
     PRIMARY KEY(albid)
)

CREATE TABLE al_include(
     sid CHAR(10) NOT NULL,
     albid CHAR(10),
     order CHAR(20),
     PRIMARY KEY (sid, albid),
     FOREIGN KEY (albid) REFERENCES Albums,
     FOREIGN KEY (sid) REFERENCES Songs
)

CREATE TABLE ua_like(
     uid CHAR(10),
     albid CHAR(10),
     PRIMARY KEY (uid, albid),
     FOREIGN KEY (uid) REFERENCES Users,
     FOREIGN KEY (albid) REFERENCES Albums
)

CREATE TABLE us_like(
     uid CHAR(10),
     sid CHAR(10),
     PRIMARY KEY (uid, sid),
     FOREIGN KEY (uid) REFERENCES Users,
     FOREIGN KEY (sid) REFERENCES Songs
)

CREATE TABLE played(
     sid CHAR(10),
     uid CHAR(10) NOT NULL AND UNIQUE,
     cnt INTEGER,
     sno INTEGER,
     PRIMARY KEY (uid, sno, sid),
     FOREIGN KEY (sid) REFERENCES Song,
     FOREIGN KEY (uid,sno) REFERENCES Sessions
          ON DELETE CASCADE)

CREATE TABLE Sessions_sesof (
     sno INTEGER,
     sstart CHAR(10),
     send CHAR(10),
     uid INTEGER NOT NULL,
     PRIMARY KEY (sno, uid),
     FOREIGN KEY (uid) REFERENCES Users
          ON DELETE CASCADE)

CREATE TABLE pub_playlists(
     plid CHAR(20),
     title CHAR(20),
     PRIMARY KEY (plid),
     FOREIGN KEY (plid) REFERENCES Playlists
)

CREATE TABLE up_like(
     uid CHAR(10),
     plid CHAR(20),
     PRIMARY KEY (uid, plid),
     FOREIGN KEY (uid) REFERENCES Users,
     FOREIGN KEY (plid) REFERENCES pub_playlist)
)
